**Goal:** Implement Phase 1 backend foundations for schema, validation, link creation service, public redirect behavior, and privacy-conscious click event insertion.

**Architecture:** Build the Phase 1 vertical slice around `GET /:code`: validate and persist link data, resolve effective link state, return deterministic redirect statuses, and insert analytics events as best effort. export function normalizeAlias(input: string): string {
  return input.trim().toLowerCase();
}

export function validateAliasOrThrow(input: string): string {
  const normalized = normalizeAlias(input);

  if (!ALIAS_REGEX.test(normalized)) {
    throw new Error("validation_error");
  }

  if (isReservedRouteSegment(normalized)) {
    throw new Error("reserved_alias");
  }

  return normalized;

const ALLOWED_SCHEMES = new Set(["http:", "https:"]);

export function validateDestinationUrlOrThrow(rawUrl: string): URL {
  let parsed: URL;

  try {
    parsed = new URL(rawUrl);
  } catch {
    throw new Error("invalid_url");
  }

  if (!ALLOWED_SCHEMES.has(parsed.protocol)) {
    throw new Error("unsafe_url");
  }

  if (!parsed.hostname || parsed.username || parsed.password) {
    throw new Error("unsafe_url");
  }

  // Reject known local/private/metadata literal IP destinations here.
  if (isUnsafeLiteralIp(parsed.hostname) || isUnsafeHostname(parsed.hostname)) {
    throw new Error("unsafe_url");
  }

  return parsed;
}
```

export function normalizeHostForBlocklist(host: string): string {
  return host.toLowerCase().replace(/\.$/, "");
}

export function domainMatchesBlocklist(
  host: string,
  rows: Array<{ domain: string; disabledAt: Date | null }>,
): boolean {
  const normalizedHost = normalizeHostForBlocklist(host);

  return rows.some((row) => {
    if (row.disabledAt) return false;

    const blocked = normalizeHostForBlocklist(row.domain);
    return normalizedHost === blocked || normalizedHost.endsWith(`.${blocked}`);
  });
}
```

export function resolveLinkState(
  link: {
    status: "active" | "disabled" | "flagged" | "deleted";
    deletedAt: Date | null;
    expiresAt: Date | null;
  },
  now: Date,
): "deleted" | "flagged" | "disabled" | "expired" | "active" {
  if (link.deletedAt || link.status === "deleted") return "deleted";
  if (link.status === "flagged") return "flagged";
  if (link.status === "disabled") return "disabled";
  if (link.expiresAt && link.expiresAt <= now) return "expired";
  return "active";
}
```

export async function resolveRedirect(input: { code: string }) {
  const link = await findByCode(input.code.toLowerCase());
  if (!link) return { status: 404 as const };

  const state = resolveLinkState(link, new Date());
  if (state === "deleted") return { status: 404 as const };
  if (state !== "active") return { status: 410 as const };

  await Promise.race([
    insertClickEvent(sanitizeClickEvent(...)),
    new Promise((_, reject) => setTimeout(() => reject(new Error("analytics_timeout")), 50)),
  ]).catch(() => undefined);

  return { status: 302 as const, location: link.destinationUrl };
}
```

app.get("/:code", async ({ params, set, request }) => {
  if (isReservedRouteSegment(params.code)) {
    set.status = 404;
    return { error: "not_found" };
  }

  const decision = await resolveRedirect({ code: params.code, request });
  if (decision.status === 302) {
    set.status = 302;
    set.headers.Location = decision.location;
    return;
  }

  set.status = decision.status;
  return { error: "not_found" };
});
